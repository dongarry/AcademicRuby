  def <=> card
    if rank_of_kind > card.rank_of_kind
      return 1
    end
    if rank_of_kind < card.rank_of_kind
      return -1
    end
    if rank_of_suit > card.rank_of_suit
      return 1
    end
    if rank_of_suit < card.rank_of_suit
      return -1
    end
    if rank_of_suit == card.rank_of_suit
      return 0 # should not happen in a single-pack card game!
    end
  end
