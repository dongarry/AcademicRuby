  def play
    puts @p1.to_string
    puts @p2.to_string

    puts "\nGame about to start!\n"

    t1 = Thread.new do
      @hand_size.times do
        card = @p1.play
        puts "#{@p1.name} plays #{card.to_string} "
        @the_pile.add card
        Kernel.sleep 0.5
        if @the_pile.snap?
          puts "#{@p1.name} shouts SNAP!!!! "
        end
      end
    end

    t2 = Thread.new do
      @hand_size.times do
        card = @p2.play
        puts "#{@p2.name} plays #{card.to_string} "
        @the_pile.add card
        Kernel.sleep 0.5
        if @the_pile.snap?
          puts "#{@p2.name} shouts SNAP!!!! "
        end
      end
    end

    t1.join
    t2.join

    puts @p1.to_string
    puts @p2.to_string
  end
