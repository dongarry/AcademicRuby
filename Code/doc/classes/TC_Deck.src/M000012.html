  def test_add_and_remove
    my_deck = Deck.new
    ace_of_spades = Card.new :SPADES, :ACE
    assert(my_deck.has_card?(ace_of_spades))
    my_deck.remove ace_of_spades
    assert_equal(my_deck.has_card?(ace_of_spades), false)
    my_deck.add ace_of_spades
    assert(my_deck.has_card?(ace_of_spades))
  end
