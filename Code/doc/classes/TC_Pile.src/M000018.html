  def test_winner
    # create pile with entire pack of cards
    SUITS.each do |suit|
      KINDS.each do |kind|
        card = Card.new suit, kind
        @pile.add card
      end
    end
    # check that ace of spades is the winner
    ace_of_spades = Card.new :SPADES, :ACE
    assert_equal(ace_of_spades, @pile.calculate_winner)
  end
